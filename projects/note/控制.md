def rate_lim_pos(self, curr_cmd, prev_cmd, rate_lim_pos, dt):
    limited_cmd = np.copy(curr_cmd)
    if (curr_cmd - prev_cmd) > rate_lim_pos*dt:
        limited_cmd = prev_cmd + rate_lim_pos*dt
    return limited_cmd
